# 配置规范

// 这个文件描述标注管理微件配置信息
{
"markers":[
  // 下面是单个标注
  {
    // 标注id
    "id": "b71350b4-6f9b-42ec-9f16-3323e47a7510",
    // 标注标题
    "title": "XX",
    // 标注描述
    "description": "XXX",
    // 标注中心点,[number, number]
    "center": [116, 40],
    // 标注附加的覆盖物，符合GeoJSON规范
    "feature": {
      "id": "2ab9215039e351945d12d702d458ee42",
      "type": "Feature",
      "properties": {},
      "geometry": {
        "coordinates": [116, 40],
        "type": "Point"
      }
    },
    // 标注附加的图片路径，可通过上传设置
    "picture": ""
  }
  //...
],
// 设置标注的显示方式，默认为default，此时标注范围常亮；可修改为hover，此时标注范围为鼠标悬停显示
"markerShowType":"default",
// 设置标注微件打开时鼠标的样式，默认为default箭头显示，可修改为grab以手型显示
"cursorType":"default"
}
// 由一组标注组成
