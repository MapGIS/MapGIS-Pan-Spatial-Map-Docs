# 行政区划配置

# 配置规范

// 这个文件描述行政区划配置信息
// 整个配置以年度为单位，是一个数组，每个年度可以配置首级信息和它下面所有其他级别的信息
[
  // 下面表示一个年度的配置信息
  {
    // 年度信息，可根据它在所有年度中进行对比找到最新年度信息
    "year": "2017",
    // 首级行政区划配置信息，包含公共的数据获取配置
    "defaults": {
      // 区划名称
      "text": "XX市",
      // 区划级别
      "level": "3",
      // 区划编码
      "code": "511500",
      // 数据获取服务器ip
      "ip": "192.168.27.246",
      // 数据获取服务器端口
      "port": "6163",
      // 数据获取方式（gdbp：矢量图层；doc：地图文档）
      "queryWay": "gdbp",
      // 地图文档名称，当queryWay为doc时有效
      "docName": ""
    },
    // 下级行政区划配置，是一个数组结构，首级下面的所有级别都包含在里面，他们之间通过level进行区分层级
    "gdbpInfos": [
      // 某个下级行政区划配置
      {
        // 行政区划级别
        "level": "4",
        // 矢量图层地址，当上面的queryWay为gdbp时有效
        "gdbp": "gdbp://MapGISLocal/index/sfcls/X511500JJQH2017GXIANXZQ",
        // 地图文档图层地址，当上面的queryWay为doc时有效
        "layerName": "",
        // 行政区划代码字段
        "codeField": "XZQDM",
        // 行政区划名称字段
        "nameField": "XZQMC"
      }
    ]
  }
]

# 行政区划规范

level 共 6 个级别

级别 含义
1 国家
2 省（自治区、直辖市、特别行政区）
3 市（地区、自治州、盟及国家直辖市所属市辖区和县的汇总码）
4 县（市辖区、县级市、旗）
5 乡镇街道
6 村、居委会

code 共 12 位

省  市  县  镇  村
XX  XX  XX  XXX XXX
1-2 3-4 5-6 7-9 10-12
级别 位数 含义
2 2 省（自治区、直辖市、特别行政区）
3 2 市（地区、自治州、盟及国家直辖市所属市辖区和县的汇总码）
4 2 县（市辖区、县级市、旗）
5 3 乡镇街道
6 3 村、居委会

提示

110101001001 北京市 市辖区 东城区 东华门街道办事处 多福巷社区居委会

420111080001 湖北省 武汉市 洪山区 关东街道办事处 关东社区居委会
